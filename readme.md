SimpleBlog
==========
Educational mini project

Tech Stack: Java (javaee, servlet, jsp, jdbc), Maven, Tomcat, sqlite

Installation
------------

Database
--------
CREATE TABLE BLOG(
ROWID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
AUTHOR TEXT NOT NULL,
TEXT TEXT NOT NULL);

CREATE TABLE USERS(
USER TEXT NOT NULL,
PASSWORD TEXT NOT NULL);
